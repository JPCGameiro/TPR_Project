    -> python3 basePktSampling_Remastered.py -i Captures/client2.pcapng -c 192.168.1.0/24 -s 0.0.0.0/0 -o client2_d1.dat
    -> python3 basePktSampling_Remastered.py -i Captures/client3.pcapng -c 192.168.1.0/24 -s 0.0.0.0/0 -o client3_d1.dat
    -> python3 basePktSampling_Remastered.py -i Captures/attacker.pcapng -c 192.168.1.0/24 -s 0.0.0.0/0 -o attacker_d1.dat
    -> python3 basePktSampling_Remastered.py -i Captures/goodAndBad.pcapng -c 192.168.1.0/24 -s 0.0.0.0/0 -o goodAndBad_d1.dat


2) Sliding Window (size of 120 seconds, 20 second period for decisions)
    -> cd Captures && python3 ../baseObsWindows_Remastered.py -i client1_d1.dat -w 120 -s 20 && cd ..
    -> cd Captures && python3 ../baseObsWindows_Remastered.py -i client2_d1.dat -w 120 -s 20 && cd ..
    -> cd Captures && python3 ../baseObsWindows_Remastered.py -i client3_d1.dat -w 120 -s 20 && cd ..
    -> cd Captures && python3 ../baseObsWindows_Remastered.py -i attacker_d1.dat -w 120 -s 20 && cd ..
    -> python3 baseObsWindows_Remastered.py -i goodAndBad_d1.dat -w 120 -s 20
    -> generates client1_d1_obs_s20, client2_d1_obs_s20, client3_d1_obs_s20, attacker_d1_obs_s20, goodAndBad_d1_obs_s20

    -> python3 baseExtractFeatures_Remastered.py -i Captures/client3_d1_obs_s20 -w 120
    -> python3 baseExtractFeatures_Remastered.py -i Captures/attacker_d1_obs_s20 -w 120
